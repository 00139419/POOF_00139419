CREATE TABLE DEPARTAMENTO(
	idDepartamento SERIAL PRIMARY KEY,
	Nombre VARCHAR(40) NOT NULL,
	Ubicacion VARCHAR(40) NOT NULL
);

CREATE TABLE USUARIO(
	idUsuario SERIAL PRIMARY KEY,
	NombreUsuario VARCHAR(50) NOT NULL,
	ApellidoUsuario VARCHAR(50) NOT NULL,
	Contrasenia VARCHAR(20) NOT NULL,
	Dui VARCHAR(17) UNIQUE NOT NULL ,
	Nacimiento VARCHAR(50) NOT NULL,
	IdDepto INT NOT NULL
);

ALTER TABLE USUARIO ADD CONSTRAINT fk_usuarioxdepartamento 
FOREIGN KEY(IdDepto)REFERENCES DEPARTAMENTO(idDepartamento)


CREATE TABLE REGISTRO(
	idRegistro SERIAL PRIMARY KEY,
	Entrada boolean NOT NULL,
	Fecha VARCHAR(50) NOT NULL,
	Temperatura FLOAT(10) NOT NULL,
	IdUsuario INT NOT NULL
);
ALTER TABLE REGISTRO ADD CONSTRAINT fk_registroxusuario 
FOREIGN KEY(IdUsuario)REFERENCES USUARIO(idUsuario)
